# 2. HELPER FUNCTIONS
# ==========================================
def calculate_grade(total_points):
    """
    Calculates Job Grade (I to XX) based on total points.
    Grade I: 0-100. Grade XX: 1401-1500.
    18 Grades in between distributed at ~72 point increments.
    """
    if total_points <= 100: return "I"
    if total_points >= 1401: return "XX"
    
    # 72 point increments starting from 101
    grade_thresholds = [
        (100, "I"), (172, "II"), (244, "III"), (316, "IV"), (388, "V"),
        (460, "VI"), (532, "VII"), (604, "VIII"), (676, "IX"), (748, "X"),
        (820, "XI"), (892, "XII"), (964, "XIII"), (1036, "XIV"), (1108, "XV"),
        (1180, "XVI"), (1252, "XVII"), (1324, "XVIII"), (1400, "XIX"), (1500, "XX")
    ]
    
    for i in range(len(grade_thresholds) - 1):
        if total_points > grade_thresholds[i][0] and total_points <= grade_thresholds[i+1][0]:
            return grade_thresholds[i+1][1]
    
    return "Unknown"

def generate_csv(data_dict):
    """Generates a CSV file from the evaluation dictionary."""
    df = pd.DataFrame([data_dict])
    return df.to_csv(index=False).encode('utf-8')

# ==========================================
# 3. STREAMLIT FRONTEND / APP STRUCTURE
# ==========================================
st.set_page_config(page_title="Job Evaluation Tool", page_icon="⚖️", layout="wide")

st.title("⚖️ Corporate Job Evaluation Tool")
st.markdown("Determine accurate job grades using the standardized Point Factor Method.")

# --- SIDEBAR: Metadata ---
st.sidebar.header("📋 Job Information")
job_title = st.sidebar.text_input("Job Title", placeholder="e.g., Senior Software Engineer")
department = st.sidebar.text_input("Department", placeholder="e.g., Engineering")
evaluator = st.sidebar.text_input("Evaluator Name", placeholder="e.g., Jane Doe, HR")

st.sidebar.markdown("---")
st.sidebar.info("Select the appropriate level for each of the 10 factors in the main panel. The total score and job grade will calculate automatically.")

# --- INITIALIZE SESSION STATE FOR SCORES ---
if 'factor_scores' not in st.session_state:
    st.session_state.factor_scores = {factor: 0 for factor in FACTOR_PLAN.keys()}
if 'factor_selections' not in st.session_state:
    st.session_state.factor_selections = {factor: "" for factor in FACTOR_PLAN.keys()}

# --- MAIN PANEL: Live Dashboard ---
total_score = sum(st.session_state.factor_scores.values())
predicted_grade = calculate_grade(total_score)

dash_col1, dash_col2, dash_col3 = st.columns(3)
with dash_col1:
    st.metric(label="Total Points (Max 1500)", value=f"{total_score} pts")
with dash_col2:
    st.metric(label="Predicted Job Grade", value=f"Grade {predicted_grade}")
with dash_col3:
    progress_pct = min(total_score / 1500.0, 1.0)
    st.progress(progress_pct, text="Scoring Progress")

st.markdown("---")

# --- MAIN PANEL: Factor Selection Tabs ---
tabs = st.tabs(list(FACTOR_PLAN.keys()))

for i, (factor_name, factor_data) in enumerate(FACTOR_PLAN.items()):
    with tabs[i]:
        st.subheader(f"{factor_name} (Max {factor_data['max_points']} pts)")
        
        # Create a list of options formatted nicely
        options = [lvl["level"] for lvl in factor_data["levels"]]
        
        # Dropdown selection
        selected_level = st.selectbox(
            f"Select level for {factor_name}:",
            options=options,
            key=f"select_{factor_name}"
        )
        
        # Find the selected item dictionary
        selected_item = next(item for item in factor_data["levels"] if item["level"] == selected_level)
        
        # Update session state (which triggers a rerun naturally in Streamlit)
        st.session_state.factor_scores[factor_name] = selected_item["points"]
        st.session_state.factor_selections[factor_name] = selected_level
        
        # Display the definition block and points
        st.info(f"**Definition:** {selected_item['desc']}")
        st.success(f"**Assigned Points:** {selected_item['points']}")

# --- FOOTER: Summary & Export ---
st.markdown("---")
st.header("📄 Final Evaluation Report")

if job_title:
    st.write(f"**Evaluating:** {job_title} | **Dept:** {department} | **By:** {evaluator}")

# Prepare data for display and export
report_data = {
    "Job Title": job_title,
    "Department": department,
    "Evaluator": evaluator,
    "Total Score": total_score,
    "Job Grade": predicted_grade
}
# Append factor details
for factor, level in st.session_state.factor_selections.items():
    report_data[factor] = level
    report_data[f"{factor} Points"] = st.session_state.factor_scores[factor]

# Display as a clean table
df_report = pd.DataFrame(list(st.session_state.factor_selections.items()), columns=["Factor", "Selected Level"])
df_report["Points"] = df_report["Factor"].map(st.session_state.factor_scores)

col_a, col_b = st.columns([2, 1])
with col_a:
    st.dataframe(df_report, use_container_width=True, hide_index=True)
with col_b:
    csv = generate_csv(report_data)
    st.download_button(
        label="⬇️ Download CSV Report",
        data=csv,
        file_name=f"Job_Eval_{job_title.replace(' ', '_') if job_title else 'Draft'}.csv",
        mime='text/csv',
        use_container_width=True
    )
